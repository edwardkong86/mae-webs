## Install

```bash
npm install --save mae-webs
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'mae-webs'
import 'mae-webs/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
